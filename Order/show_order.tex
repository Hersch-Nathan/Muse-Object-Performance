\newcommand{\ObjectCell}[1]{%
  \ifthenelse{\equal{#1}{Shirt}}{\cellcolor{shirtcolor}#1}{%
    \ifthenelse{\equal{#1}{Muppet}}{\cellcolor{muppetcolor}#1}{%
      \ifthenelse{\equal{#1}{Animatronic}}{\cellcolor{animatroniccolor}#1}{%
        \ifthenelse{\equal{#1}{Robot}}{\cellcolor{robotcolor}#1}{#1}%
      }%
    }%
  }%
}

\newcommand{\PerformerCell}[1]{%
  \ifthenelse{\equal{#1}{Moose}}{\cellcolor{moosecolor}#1}{%
    \ifthenelse{\equal{#1}{Luca}}{\cellcolor{lucacolor}#1}{%
      \ifthenelse{\equal{#1}{None}}{\cellcolor{nonecolor}#1}{#1}%
    }%
  }%
}
